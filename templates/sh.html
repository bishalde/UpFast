1.	Greater of 3 numbers :

echo "enter 1st number" read num1
echo "enter 2nd number" read num2
echo "enter 3rd number" read num3
greater=$num1
if [ $greater -lt $num2 ] then
greater=$num2
fi
if [ $greater -lt $num3 ] then
greater=$num3
fi
echo "The greater of 3 number is $greater"

5.	Sum of digits of number
echo "enter number"
read num
sum=0
while [ $num -gt 0 ]
do
sum=$((sum+=num%10)) num=$((num/10))
done
echo "sum of n odd numbers are $sum"


6.	Factorial of number :
echo "enter number"
read num
fact=1
while [ $num -gt 0 ] do
((fact*=num)) ((num--))
done
echo "$fact"



7.	Sum of array element
declare -a arr=( 10 20 30 )
sum=0
for i in ${arr[@]}
do
sum=$(($sum+i))
done
echo "sum of array elements : $sum"
